# Telegram Markdown text

**Python –º–æ–¥—É–ª—å, —â–æ –±—É–≤ —Å—Ç–≤–æ—Ä–µ–Ω–∏–π –¥–ª—è —Å–ø—Ä–æ—â–µ–Ω–Ω—è —Ä–æ–±–æ—Ç–∏ –∑ MarkdownV2 –ø—Ä–∏ –Ω–∞–ø–∏—Å–∞–Ω–Ω—ñ —Ç–µ–ª–µ–≥—Ä–∞–º –±–æ—Ç–∞. –û—Å–Ω–æ–≤–Ω–æ—é —ñ–¥–µ—î—é –±—É–ª–æ
—Å—Ç–≤–æ—Ä–∏—Ç–∏ —â–æ—Å—å —Å—Ö–æ–∂–µ –Ω–∞ StringBuilder –∑ –º–æ–≤–∏ –ø—Ä–æ–≥—Ä–∞–º—É–≤–∞–Ω–Ω—è Kotlin.**

### –£—Å—Ç–∞–Ω–æ–≤–∫–∞

–í—Å—Ç–∞–Ω–æ–≤–∏—Ç–∏ –º–æ–∂–Ω–∞ –∫–æ–º–∞–Ω–¥–æ—é `pip install telegram-markdown-text`.

### –Ø–∫ —é–∑–∞—Ç—å?

–ú–æ–¥—É–ª—å –ø—ñ–¥—Ç—Ä–∏–º—É—î —É—Å—ñ –¥–æ—Å—Ç—É–ø–Ω—ñ —Å—Ç–∏–ª—ñ –∑ `MarkdownV2` –æ–ø–∏—Å–∞–Ω—ñ
–≤ [–æ—Ñ—ñ—Ü—ñ–π–Ω—ñ–π –¥–æ–∫—É–º–µ–Ω—Ç–∞—Ü—ñ—ó](https://core.telegram.org/bots/api#markdownv2-style)

```python
from telegram_markdown_text import *

Bold('–ñ–∏—Ä–Ω–∏–π —Ç–µ–∫—Å—Ç')
# *–ñ–∏—Ä–Ω–∏–π —Ç–µ–∫—Å—Ç*

Italic('–ö—É—Ä—Å–∏–≤')
# _–ö—É—Ä—Å–∏–≤_

Underline('–ü—ñ–¥–∫—Ä–µ—Å–ª–µ–Ω–∏–π —Ç–µ–∫—Å—Ç')
# __–ü—ñ–¥–∫—Ä–µ—Å–ª–µ–Ω–∏–π —Ç–µ–∫—Å—Ç__

Strikethrough('–ó–∞–∫—Ä–µ—Å–ª–µ–Ω–∏–π —Ç–µ–∫—Å—Ç')
# ~–ó–∞–∫—Ä–µ—Å–ª–µ–Ω–∏–π —Ç–µ–∫—Å—Ç~

Spoiler('–°–ø–æ–π–ª–µ—Ä')
# ||–°–ø–æ–π–ª–µ—Ä||

InlineUrl('–Ø–∫–∞—Å—å –ª—ñ–Ω–∫–∞', 'https://test.url')
# [–Ø–∫–∞—Å—å –ª—ñ–Ω–∫–∞](https://test.url)

InlineUser('–õ—ñ–Ω–∫–∞ –Ω–∞ —é–∑–µ—Ä–∞', 123456)
# [–õ—ñ–Ω–∫–∞ –Ω–∞ —é–∑–µ—Ä–∞](tg://user?id=123456)

Emoji('üëç', 123456)
# [üëç](tg://emoji?id=123456)

InlineCode('–í—ñ–¥—Ñ–æ—Ä–º–∞—Ç–æ–≤–∞–Ω–∏–π —Ä—è–¥–æ–∫ –∫–æ–¥—É')
# `–í—ñ–¥—Ñ–æ—Ä–º–∞—Ç–æ–≤–∞–Ω–∏–π —Ä—è–¥–æ–∫ –∫–æ–¥—É`

InlineCodeBlock('–í—ñ–¥—Ñ–æ—Ä–º–∞—Ç–æ–≤–∞–Ω–∏–π –±–ª–æ–∫ –∫–æ–¥—É –∑ –ø—ñ–¥—Å–≤—ñ—Ç–∫–æ—é —Å–∏–Ω—Ç–∞–∫—Å–∏—Å—É', 'kotlin')
# ```kotlin
# –í—ñ–¥—Ñ–æ—Ä–º–∞—Ç–æ–≤–∞–Ω–∏–π –±–ª–æ–∫ –∫–æ–¥—É –∑ –ø—ñ–¥—Å–≤—ñ—Ç–∫–æ—é —Å–∏–Ω—Ç–∞–∫—Å–∏—Å—É
# ````

QuoteBlock('–¶–∏—Ç–∞—Ç–∞')
# ```kotlin
# >–¶–∏—Ç–∞—Ç–∞**
# ````

PlainText('–ü—Ä–æ—Å—Ç–æ —Ç–µ–∫—Å—Ç')
# –ü—Ä–æ—Å—Ç–æ —Ç–µ–∫—Å—Ç
```

–í–∞—Ä—ñ–∞–Ω—Ç –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è `MarkdownText` –∑ `append()`

```python
from telegram_markdown_text import *

text = MarkdownText()
text.append('–ü—Ä–æ—Å—Ç–∏–π —Ç–µ–∫—Å—Ç ')
text.append(Bold('–∑ –∂–∏—Ä–Ω–∏–º–∏ —Ç–µ–∫—Å—Ç–æ–º'))
text.append(Italic(' —Ç–∞ –∫—É—Ä—Å–∏–≤–æ–º'))
print(text)

# –ü—Ä–æ—Å—Ç–∏–π —Ç–µ–∫—Å—Ç *–∑ –∂–∏—Ä–Ω–∏–º —Ç–µ–∫—Å—Ç–æ–º* —Ç–∞ –∫—É—Ä—Å–∏–≤–æ–º
```

–í–∞—Ä—ñ–∞–Ω—Ç –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è –∑ –¥–æ–¥–∞–≤–∞–Ω–Ω—è–º `+`

```python
from telegram_markdown_text import *

text = PlainText('–ü—Ä–æ—Å—Ç–∏–π —Ç–µ–∫—Å—Ç ') + Bold('–∑ –∂–∏—Ä–Ω–∏–º–∏ —Ç–µ–∫—Å—Ç–æ–º') + Italic(' —Ç–∞ –∫—É—Ä—Å–∏–≤–æ–º')
print(text)

# –ü—Ä–æ—Å—Ç–∏–π —Ç–µ–∫—Å—Ç *–∑ –∂–∏—Ä–Ω–∏–º —Ç–µ–∫—Å—Ç–æ–º* —Ç–∞ –∫—É—Ä—Å–∏–≤–æ–º
```

–í–∞—Ä—ñ–∞–Ω—Ç –≤–∫–ª–∞–¥–∞–Ω–Ω—è —Ç–µ–∫—Å—Ç—É –æ–¥–Ω–æ–≥–æ —Å—Ç–∏–ª—é –≤ —ñ–Ω—à–∏–π

```python
from telegram_markdown_text import *

text = Bold((Italic('–ñ–∏—Ä–Ω–∏–π –∫—É—Ä—Å–∏–≤') + ' —Ç–∞ ' + Underline('–∂–∏—Ä–Ω–∏–π –∞–Ω–¥–µ—Ä–ª–∞–π–Ω')))
print(text)

# *–ñ–∏—Ä–Ω–∏–π –∫—É—Ä—Å–∏–≤ —Ç–∞ __–∂–∏—Ä–Ω–∏–π –∞–Ω–¥–µ—Ä–ª–∞–π–Ω__*
```

–Ø–∫—â–æ `.append()` –≤–∏–∫–ª–∏–∫–∞—Ç–∏ –Ω–∞ —è–∫–æ–º—É—Å—å –∫–æ–Ω–∫—Ä–µ—Ç–Ω–æ–º—É –µ–ª–µ–º–µ–Ω—Ç—ñ, –∞ –Ω–µ –Ω–∞ `MarkdownText`, —Ç–æ –¥–æ–¥–∞–Ω–∏–π —Ç–µ–∫—Å—Ç –ø—Ä–æ—Å—Ç–æ —Å—Ç–∞–Ω–µ –π–æ–≥–æ
—á–∞—Å—Ç–∏–Ω–æ—é —ñ –ø–æ–≤—Ç–æ—Ä–∏—Ç—å –π–æ–≥–æ —Å—Ç–∏–ª—å

```python
from telegram_markdown_text import *

text = Bold('–ñ–∏—Ä–Ω–∏–π —Ç–µ–∫—Å—Ç').append(' —Ç–∞ —Ç–∞–∫–∏–π –∂–µ –∂–∏—Ä–Ω–∏–π —Ç–µ–∫—Å—Ç')
print(text)

# *–ñ–∏—Ä–Ω–∏–π —Ç–µ–∫—Å—Ç —Ç–∞ —Ç–∞–∫–∏–π –∂–µ –∂–∏—Ä–Ω–∏–π —Ç–µ–∫—Å—Ç*
```

### –¶–∏—Ç–∞—Ç–∏

–ß–µ—Ä–µ–∑ –æ—Å–æ–±–ª–∏–≤–æ—Ç—Å—ñ Telegram API, —Ü–∏—Ç–∞—Ç–∏ –º–∞—é—Ç—å –ø–µ–≤–Ω—ñ –æ—Å–æ–±–ª–∏–≤–æ—Å—Ç—ñ –≤ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—ñ:

1. –î–æ–¥–∞–≤–∞—Ç–∏ —Ü–∏—Ç–∞—Ç—É —Ç—Ä–µ–±–∞ –ª–∏—à–µ –∑ –Ω–æ–≤–æ–≥–æ —Ä—è–¥–∫–∞. –Ø–∫—â–æ –¥–æ–¥–∞—Ç–∏ —Ü–∏—Ç–∞—Ç—É –Ω–µ –∑ –Ω–æ–≤–æ–≥–æ —Ä—è–¥–∫–∞ - —Ç–µ–ª–µ–≥—Ä–∞–º –∫–∏–Ω–µ –ø–æ–º–∏–ª–∫—É
2. –î–æ–¥–∞–≤–∞—Ç–∏ —Ç–µ–∫—Å—Ç –ø—ñ—Å–ª—è —Ü–∏—Ç–∞—Ç–∏ —Ç–∞–∫–æ–∂ –ø–æ—Ç—Ä—ñ–±–Ω–æ –¥–æ–¥–∞–≤–∞—Ç–∏ –ª–∏—à–µ –∑ –Ω–æ–≥–æ —Ä—è–¥–∫–∞. –Ø–∫—â–æ –¥–æ–¥–∞—Ç–∏ —Ç–µ–∫—Å—Ç –±–µ–∑ –Ω–æ–≤–æ–≥–æ —Ä—è–¥–∫–∞ - —Ü–µ–π —Ç–µ–∫—Å—Ç
   —Å—Ç–∞–Ω–µ —á–∞—Å—Ç–∏–Ω–æ—é —Ü–∏—Ç–∞—Ç–∏
3. –ü—Ä–∏ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—ñ –¥–≤–æ—Ö —Ü–∏—Ç–∞—Ç –ø—ñ–¥—Ä—è–¥ - –ø–µ—Ä–µ–Ω–µ—Å–µ–Ω–Ω—è –Ω–∞ –Ω–æ–≤–∏–π —Ä—è–¥–æ–∫ –ø–æ–≤–∏–Ω–Ω–æ –±—É—Ç–∏ —á–∞—Å—Ç–∏–Ω–æ—é –ø–æ–ø–µ—Ä–µ–¥–Ω—å–æ—ó —Ü–∏—Ç–∞—Ç–∏

–í–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è —Ü–∏—Ç–∞—Ç–∏ –≤ —è–∫–æ—Å—Ç—ñ –æ–∫—Ä–µ–º–æ–≥–æ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è

```python
from telegram_markdown_text import *

text = QuoteBlock("–ü—Ä–æ—Ü–∏—Ç–æ–≤–∞–Ω–∏–π —Ç–µ–∫—Å—Ç\n–≤ –¥–≤–∞ —Ä—è–¥–∫–∏")
```

–ü—Ä–∞–≤–∏–ª—å–Ω–µ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è —Ü–∏—Ç–∞—Ç–∏ –≤ —Å–µ—Ä–µ–¥–∏–Ω—ñ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è

```python
from telegram_markdown_text import *

text = MarkdownText()
text.append('–ü–æ—á–∞—Ç–æ–∫ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è')
text.append_ln()  # –≤–∞–∂–ª–∏–≤–æ –¥–æ–¥–∞—Ç–∏ –ø–µ—Ä–µ–Ω–µ—Å–µ–Ω–Ω—è –Ω–∞ –Ω–æ–≤–∏–π —Ä—è–¥–æ–∫ –ø–µ—Ä–µ–¥ —Ü–∏—Ç–∞—Ç–æ—é
text.append(QuoteBlock("–ü—Ä–æ—Ü–∏—Ç–æ–≤–∞–Ω–∏–π —Ç–µ–∫—Å—Ç\n–≤ –¥–≤–∞ —Ä—è–¥–∫–∏"))
text.append_ln()  # –≤–∞–∂–ª–∏–≤–æ –¥–æ–¥–∞—Ç–∏ –ø–µ—Ä–µ–Ω–µ—Å–µ–Ω–Ω—è –Ω–∞ –Ω–æ–≤–∏–π —Ä—è–¥–æ–∫ –ø—ñ—Å–ª—è —Ü–∏—Ç–∞—Ç–∏
text.append('–ö—ñ–Ω–µ—Ü—å –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è')
```

–ü—Ä–∞–≤–∏–ª—å–Ω–µ –≤–∏–∫–æ—Ä–∏—Å—Ç–∞–Ω–Ω—è –¥–µ–∫—ñ–ª—å–∫–æ—Ö —Ü–∏—Ç–∞—Ç –ø—ñ–¥—Ä—è–¥ –≤ —Å–µ—Ä–µ–¥–∏–Ω—ñ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è

```python
from telegram_markdown_text import *

text = MarkdownText()
text.append('–ü–æ—á–∞—Ç–æ–∫ –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è')
text.append_ln()  # –≤–∞–∂–ª–∏–≤–æ –¥–æ–¥–∞—Ç–∏ –ø–µ—Ä–µ–Ω–µ—Å–µ–Ω–Ω—è –Ω–∞ –Ω–æ–≤–∏–π —Ä—è–¥–æ–∫ –ø–µ—Ä–µ–¥ —Ü–∏—Ç–∞—Ç–∞–º–∏
text.append(QuoteBlock("–ü–µ—Ä—à–∞ —Ü–∏—Ç–∞—Ç–∞\n"))
text.append(QuoteBlock("–î—Ä—É–≥–∞ —Ü–∏—Ç–∞—Ç–∞\n"))
text.append(QuoteBlock("–¢—Ä–µ—Ç—è —Ü–∏—Ç–∞—Ç–∞"))
text.append_ln()  # –≤–∞–∂–ª–∏–≤–æ –¥–æ–¥–∞—Ç–∏ –ø–µ—Ä–µ–Ω–µ—Å–µ–Ω–Ω—è –Ω–∞ –Ω–æ–≤–∏–π —Ä—è–¥–æ–∫ –ø—ñ—Å–ª—è —Ü–∏—Ç–∞—Ç
text.append('–ö—ñ–Ω–µ—Ü—å –ø–æ–≤—ñ–¥–æ–º–ª–µ–Ω–Ω—è')
```

### –ï–∫—Ä–∞–Ω—É–≤–∞–Ω–Ω—è Markdown —Å–∏–º–æ–≤–ª—ñ–≤

–î–ª—è —Ç–æ–≥–æ, —â–æ–± –æ—Ç—Ä–∏–º–∞—Ç–∏ –µ–∫—Ä–∞–Ω–æ–≤–∞–Ω–∏–π —Ç–µ–∫—Å—Ç, –¥–æ—Å—Ç–∞—Ç–Ω—å–æ –ø—Ä–æ—Å—Ç–æ –≤–∏–∫–ª–∏–∫–∞—Ç–∏ –º–µ—Ç–æ–¥ `escaped_text()`

```python
from telegram_markdown_text import *

text = PlainText('–ü—Ä–æ—Å—Ç–∏–π —Ç–µ–∫—Å—Ç –∑ —Ä—ñ–∑–Ω–∏–º–∏ —Å–∏–º–≤–æ–ª–∞–º–∏ !"‚Ññ;%:?*()')
text.escaped_text()  # –µ–∫—Ä–∞–Ω–æ–≤–∞–Ω–∏–π —Ç–µ–∫—Å—Ç, —è–∫–∏–π –º–æ–∂–Ω–∞ –≤—ñ–¥–ø—Ä–∞–≤–ª—è—Ç–∏ –º–µ—Å–µ–¥–∂–µ–º —á–µ—Ä–µ–∑ API –¢–µ–ª–µ–≥—Ä–∞–º–∞
```

–ê–±–æ –∂ –ø—Ä–æ—Å—Ç–æ –ø—Ä–∏–≤–µ—Å—Ç–∏ MarkdownText –¥–æ str, —è–∫ —Ü–µ —Ä–æ–±–∏—Ç—å—Å—è –ø—Ä–∏ –≤–∏–∫–ª–∏–∫—É –º–µ—Ç–æ–¥—É `print()`

```python

print(text)  # –≤–∏–≤–µ–¥–µ—Ç—å—Å—è —É–∂–µ –µ–∫—Ä–∞–Ω–æ–≤–∞–Ω–∏–π —Ç–µ–∫—Å—Ç, –æ—Å–∫—ñ–ª—å–∫–∏ –≤ MarkdownText –æ–≤–µ—Ä—Ä–∞–π–¥–Ω—É—Ç–∏–π –º–µ—Ç–æ–¥ __str__
```
